# 1. Navigate to your project folder
cd /Users/dylandwood/HonorsProject.py

# 2. Check the current Git status to see changes
git status

# 3. Stage changes (or add a specific file, e.g., loglbm.csv)
git add .
# OR, to add only the CSV file:
# git add loglbm.csv

# 4. Commit your changes with a descriptive message
git commit -m "Update CSV file reference to loglbm.csv"

# 5. Log in to Heroku (if not already logged in)
heroku login

# 6. Link your local repository to your Heroku app (replace 'climatechangedashboard' with your app name)
heroku git:remote -a climatechangedashboard

# 7. Push your changes to Heroku (use 'main' or your current branch name)
git push heroku main

# 8. Check the Heroku logs to see if the app is running correctly
heroku logs --tail --app climatechangedashboard
